      import numpy as np
def GaussianElimination(A0, V):
		N = len(V)
		# Juntar para maior eficiência
		A = np.concatenate((A0, V.reshape((N, 1))), axis=1)
		x = np.empty(N, float)

		# Eliminação de Gauss
		for i in range(N):
				A[i, i:] /= A[i, i]
				for k in range(i+1, N):
		        A[k, i:] -= A[k, i] * A[i, i:]

		# Backsubstituting
		for i in range(N-1, -1, -1):
		    x[i] = A[i, -1]
		    for k in range(i+1, N):
		        x[i] -= A[i, k] * x[k]
		return x

      def lu(a0):
    u = np.ndarray.copy(a0)
    n = len(u)
    l = np.zeros([n, n])
    positions = np.arange(n)

    for i in range(n):
        # Pivoting
        max_index = np.argmax(np.abs(u[i:, i])) + i
        if max_index != i:
            u[[i, max_index]] = u[[max_index, i]]
            l[[i, max_index]] = l[[max_index, i]]
            positions[i], positions[max_index] = positions[max_index], positions[i]

        l[i:, i] = np.ndarray.copy(u[i:, i])
        u[i, i:] /= u[i, i]
        for k in range(i+1, n):
            u[k, i:] -= u[k, i] * u[i, i:]
    return l, u, positions


def solve_lu(a0, v):
    n = len(v)
    l, u, positions = lu(a0)
    x = np.empty(n)
    y = np.empty(n)

    # Trocar posições em v
    aux = np.ndarray.copy(v)
    for i in range(n):
        aux[i] = v[positions[i]]

    # Backsubstitution de uma matriz triangular inferior
    for i in range(n):
        y[i] = aux[i]
        for k in range(0, i):
            y[i] -= y[k] * l[i, k]
        y[i] /= l[i, i]

    # Backsubstitution de uma matriz triangular superior
    for i in range(n-1, -1, -1):
        x[i] = y[i]
        for k in range(i+1, n):
            x[i] -= x[k] * u[i, k]
        x[i] /= u[i, i]
    return x

def matrix_to_banded(a, inf, sup):
    n = len(a)
    rows = inf + sup + 1
    a0 = np.zeros([rows, n])
    for r, k in enumerate(range(-inf, sup + 1)):
        if k < 0:
            a0[rows - 1 - r, :k] = np.diag(a, k)
        else:
            a0[rows - 1 - r, k:] = np.diag(a, k)

    return a0

        v[m] /= div
        for k in range(1,down+1):
            if m+k<N:

        # Now normalize the pivot row of A and subtract from lower ones
        for i in range(up):

      def qr(a):
    n = len(a)
    q = np.empty([n, n], float)
    r = np.zeros([n, n], float)

    for i in range(n):
        u = np.ndarray.copy(a[:, i])
        for k in range(i):
            qk_ai = q[:, k] @ a[:, i]
            u -= qk_ai * q[:, k]
            r[k, i] = qk_ai
        u_norm = np.linalg.norm(u)
        r[i, i] = u_norm
        q[:, i] = u / u_norm

    return q, r

      def jacobi(A, b):
    """Solves the equation Ax=b via the Jacobi iterative method."""
    x = np.zeros(len(A[0]))
    precision = 1e-6

    error = precision + 1

    # Create a vector of the diagonal elements of A
    # and subtract them from A
    d = np.diag(A)
    r = A - np.diagflat(d)

    # Iterate for N times
    while error > precision:
        x_old = x.copy()
        x = (b - r @ x) / d
        error = np.max(np.abs(x - x_old))
    return x

      def householder_elim(q, a, i_min):
    N = len(a[:, 0]) - i_min
    e0_v = np.zeros(N); e0_v[0] = 1
    u_v = np.ndarray.copy(a[i_min:, i_min])
    u_v += np.sqrt(u_v @ u_v) * e0_v
    u_v /= np.sqrt(u_v @ u_v)
    a[i_min:, i_min:] -= 2 * np.outer(u_v, np.dot(u_v, a[i_min:, i_min:]))
    q[:, i_min:] -= 2 * np.outer(q[:, i_min:] @ u_v, u_v)


def householder_qr(a):
    n = len(a[:, 0])
    r = np.ndarray.copy(a)
    q = np.eye(n)
    for i in range(n-1):
        householder_elim(q, r, i)
    return q, r
