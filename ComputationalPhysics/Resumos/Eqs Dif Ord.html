def runge_kutta_4(n):
    h = (t_max - t_min) / n  # espaçamento temporal
    t_values = np.linspace(t_min, t_max, n+1)  # lista com os valores do tempo
    r = np.array([179 * np.pi / 180, 0], float)  # condições iniciais
    r_values = np.empty([n+1, 2], float)  # array para armazenar a solução e a derivada para os diferentes t's
    for i, t in enumerate(t_values):
        r_values[i] = r  # guardar o valor calculado
        
        # calcular o valor seguinte com o método de runge-kuta-4

  • Shooting

    Para variar condições iniciais o processo é idêntico. Em vez de receber um E, recebia uma dada condição inicial. Neste caso o erro era medido a partir da diferença entre as condições fronteira (neste caso entre psi1 e psi2) em vez das próprias condições inciais.

    Outro método eficaz para este tipo de problema é o da secante (o de Newton nem sempre se tem a derivada).

    Para encontrar valores próprios, eq de 2ª ordem.

    # Pretende-se que a solução seja 0 para a última posição

def solution(E):
    n = 1000 # número de pontos
    h = (x_max - x_min) / n # espaçamento entre posições
    x_values = np.linspace(x_min, x_max, n + 1) # array com as posições
    r = np.array((0, 1), float) # condições iniciais
    r_values = np.empty((n + 1, 2), float) # array para guardar a solução e a derivada
    for i, x in enumerate(x_values):
        r_values[i] = r # guardar os resultados para a posição anterior

				# método de runge-kutta-4
        k1 = h * f(r, x, E)
        k2 = h * f(r + k1/2, x + h/2, E)
        k3 = h * f(r + k2/2, x + h/2, E)
        k4 = h * f(r + k3, x + h, E)
        r += (k1 + 2 * k2 + 2 * k3 + k4) / 6
    return r_values[-1, 0]


# Método da bisseção, binary search
def find_energy(E1, E2): # psi1 e psi 2 têm de ter sinais contrários
    precision = 1e-8 # precisão
    error = abs(E1 - E2) # erro inicial
    psi1 = solution(E1) # calcular a solução para E1
    while error > precision:
        E_average = (E1 + E2) / 2 # ponto médio
        psi_average = solution(E_average)
        if psi_average * psi1 > 0: # se psi1 e psi_averagem têm = sinal
            E1 = E_average
            psi1 = solution(E1)
        else:
            E2 = E_average
        error = abs(E1 - E2) # calcular novo erro
    return (E1 + E2) / 2 # ponto médio


# Método da secante
def find_energy2(E1, E2, f): # psi1 e psi2 escolhidos -> ver resumos secante
    p = 1e-8 # precisão
    f2 = f(E1)

    while abs(E2 - E1) > p: # mede-se o erro a partir da energias
        f1, f2 = f2, f(E2)
        error = f2 * (E2 - E1) / (f2 - f1) # caso queira medir o erro na cond fronteira
        E1, E2 = E2, E2 - error
    return E2
