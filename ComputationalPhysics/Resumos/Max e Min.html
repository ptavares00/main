def golden_ratio(x1, x4, f, p = 1):
    z = (1 + np.sqrt(5)) / 2 
    x2 = x4 - (x4 - x1) / z
    x3 = x1 + (x4 - x1) / z
    
    f1, f2, f3, f4 = f(x1), f(x2), f(x3), f(x4)
    
    while (x4 - x1) > p:
        if f(x2) > f(x3):
            x4, f4 = x3, f3
            x3, f3 = x2, f2
            x2 = x4 - (x4 - x1) / z
            f2 = f(x2)
        else:
            x1, f1 = x2, f2
            x2, f2 = x3, f3
            x3 = x1 + (x4 - x1) / z
            f3 = f(x3)
    return (x3 + x2) / 2
